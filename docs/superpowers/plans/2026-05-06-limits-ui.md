# Limits Service — UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the `ui/limits/` Flutter package with the full operator surface described in spec §11 — policy CRUD, approval queue, runtime verdict playground, ledger search, and the audit-on-limiting feed — backed by real Connect RPC against the live `apps/limits` service.

**Architecture:** Mirrors the per-domain shape of `ui/savings/`, `ui/loans/`. New Dart proto package `antinvestor_api_limits` generated via `proto/buf.gen.dart.limits.yaml`; new Flutter package `ui/limits/` (`antinvestor_ui_limits`) consuming it. Riverpod 3 providers wrapping each RPC, go_router routes, screens built on `antinvestor_ui_core`'s shared admin widgets. Connect-over-HTTP, no mocks.

**Tech Stack:** Flutter 3.11+, Dart 3.3+, `antinvestor_ui_core ^0.2.0`, `antinvestor_api_limits ^0.1.0` (new), `antinvestor_api_common ^1.52.0`, `connectrpc ^1.0.0`, `flutter_riverpod ^3.3.1`, `go_router ^17.2.0`, `protobuf ^4.2.0`, `fixnum ^1.1.0`.

**Spec:** `docs/superpowers/specs/2026-05-05-fintech-limits-design.md` §11 (UI architecture).

**Prerequisite:** Plan 2 (`milestone/limits-runtime-and-approvals` tag) must be deployed and reachable — every UI screen consumes a live RPC.

**Out of scope:** Mobile-first responsive layout (desktop-first per spec §11.6); localisation (English only v1).

```dart
// (Apache 2.0 header)
import 'package:antinvestor_api_limits/antinvestor_api_limits.dart';
import 'package:antinvestor_ui_core/api/stream_helpers.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';

import 'limits_transport_provider.dart';

class PolicySearchParams {
  final String query;
  final LimitAction action;
  final SubjectType subjectType;
  final PolicyMode mode;
  const PolicySearchParams({
    this.query = '',
    this.action = LimitAction.LIMIT_ACTION_UNSPECIFIED,
    this.subjectType = SubjectType.SUBJECT_TYPE_UNSPECIFIED,
    this.mode = PolicyMode.POLICY_MODE_UNSPECIFIED,
  });

  @override
  bool operator ==(Object other) =>
      other is PolicySearchParams &&
      other.query == query &&
      other.action == action &&
      other.subjectType == subjectType &&
      other.mode == mode;

  @override
  int get hashCode => Object.hash(query, action, subjectType, mode);
}

/// Streaming search of policies. Cached by params for navigation back/forward.
final policyListProvider = FutureProvider.family<List<PolicyObject>, PolicySearchParams>(
  (ref, params) async {
    final client = ref.watch(limitsAdminServiceClientProvider);
    final request = PolicySearchRequest(
      query: params.query,
      action: params.action,
      subjectType: params.subjectType,
      mode: params.mode,
      cursor: PageCursor(limit: 50),
    );
    return collectStream(
      client.policySearch(request),
      extract: (response) => response.data,
    );
  },
);

/// Single policy by id.
final policyDetailProvider = FutureProvider.family<PolicyObject, String>(
  (ref, id) async {
    final client = ref.watch(limitsAdminServiceClientProvider);
    final response = await client.policyGet(PolicyGetRequest(id: id));
    return response.data;
  },
);

/// Mutation notifier. Save invalidates the list; Delete also invalidates the
/// detail of the deleted id.
class PolicyMutationNotifier extends Notifier<AsyncValue<void>> {
  @override
  AsyncValue<void> build() => const AsyncValue.data(null);

  LimitsAdminServiceClient get _client => ref.read(limitsAdminServiceClientProvider);

  Future<PolicyObject> save(PolicyObject policy) async {
    state = const AsyncValue.loading();
    try {
      final response = await _client.policySave(PolicySaveRequest(data: policy));
      state = const AsyncValue.data(null);
      ref.invalidate(policyListProvider);
      if (response.data.id.isNotEmpty) {
        ref.invalidate(policyDetailProvider(response.data.id));
      }
      return response.data;
    } catch (e, st) {
      state = AsyncValue.error(e, st);
      rethrow;
    }
  }

  Future<void> delete(String id) async {
    state = const AsyncValue.loading();
    try {
      await _client.policyDelete(PolicyDeleteRequest(id: id));
      state = const AsyncValue.data(null);
      ref.invalidate(policyListProvider);
      ref.invalidate(policyDetailProvider(id));
    } catch (e, st) {
      state = AsyncValue.error(e, st);
      rethrow;
    }
  }
}

final policyMutationProvider =
    NotifierProvider<PolicyMutationNotifier, AsyncValue<void>>(PolicyMutationNotifier.new);
```

```dart
// (Apache 2.0 header)
import 'package:antinvestor_api_limits/antinvestor_api_limits.dart';
import 'package:antinvestor_ui_core/api/stream_helpers.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';

import 'limits_transport_provider.dart';

class AuditSearchParams {
  final List<String> actions;
  final String entityType;
  final String entityId;
  final String actorId;
  const AuditSearchParams({
    this.actions = const [],
    this.entityType = '',
    this.entityId = '',
    this.actorId = '',
  });

  @override
  bool operator ==(Object other) =>
      other is AuditSearchParams &&
      _listEquals(other.actions, actions) &&
      other.entityType == entityType &&
      other.entityId == entityId &&
      other.actorId == actorId;

  @override
  int get hashCode => Object.hashAll([...actions, entityType, entityId, actorId]);
}

bool _listEquals<T>(List<T> a, List<T> b) {
  if (a.length != b.length) return false;
  for (int i = 0; i < a.length; i++) {
    if (a[i] != b[i]) return false;
  }
```

```dart
/// Format a google.protobuf.Duration as a human string like "24h", "7d", "30d".
String formatDuration(int seconds) {
  if (seconds <= 0) return '—';
  if (seconds % 86400 == 0) return '${seconds ~/ 86400}d';
  if (seconds % 3600 == 0) return '${seconds ~/ 3600}h';
  if (seconds % 60 == 0) return '${seconds ~/ 60}m';
  return '${seconds}s';
}

/// Parse a friendly duration string ("24h", "7d", "1800s", "30m") into seconds.
/// Returns null on parse error.
int? parseDuration(String s) {
  s = s.trim().toLowerCase();
  if (s.isEmpty) return null;
  final unit = s[s.length - 1];
  final n = int.tryParse(s.substring(0, s.length - 1));
  if (n == null) return null;
  switch (unit) {
    case 'd': return n * 86400;
    case 'h': return n * 3600;
    case 'm': return n * 60;
    case 's': return n;
    default:  return int.tryParse(s); // raw seconds
  }
}
```

```dart
// (Apache 2.0 header)
import 'package:flutter/material.dart';
import 'package:antinvestor_api_common/antinvestor_api_common.dart' show Money;

const _zeroDecimal = {'BIF', 'CLP', 'DJF', 'GNF', 'ISK', 'JPY', 'KMF', 'KRW',
                      'PYG', 'RWF', 'UGX', 'UYI', 'VND', 'VUV', 'XAF', 'XOF', 'XPF'};
const _threeDecimal = {'BHD', 'IQD', 'JOD', 'KWD', 'LYD', 'OMR', 'TND'};

int _decimalsFor(String code) {
  final c = code.trim().toUpperCase();
  if (_zeroDecimal.contains(c)) return 0;
  if (_threeDecimal.contains(c)) return 3;
  return 2;
}

/// Renders [Money] with the right ISO 4217 precision.
class CurrencyAmountText extends StatelessWidget {
  final Money? amount;
  final TextStyle? style;
  final bool showCurrency;

  const CurrencyAmountText({
    super.key,
    required this.amount,
    this.style,
    this.showCurrency = true,
  });

  @override
  Widget build(BuildContext context) {
    if (amount == null) return Text('—', style: style);
    final code = amount!.currencyCode;
    final decimals = _decimalsFor(code);
    final units = amount!.units.toInt();
    final nanos = amount!.nanos;

    // units + nanos*10^-9 → display with `decimals` fractional digits.
    final fractional = nanos.abs();
    final divisor = decimals == 0 ? 1 : (10 ^ (9 - decimals)); // 10^(9-decimals)
    // Use integer math to avoid float artefacts.
    final scaled = fractional ~/ _pow10(9 - decimals);
    final negative = units < 0 || nanos < 0;
    final absUnits = units.abs();

    final buffer = StringBuffer();
    if (negative) buffer.write('-');
    buffer.write(absUnits.toString());
    if (decimals > 0) {
      buffer.write('.');
      buffer.write(scaled.toString().padLeft(decimals, '0'));
    }
    if (showCurrency) {
      buffer.write(' ');
      buffer.write(code);
    }
    return Text(buffer.toString(), style: style);
  }
}

int _pow10(int n) {
  var r = 1;
  for (var i = 0; i < n; i++) {
    r *= 10;
  }
  return r;
}
```

```dart
      ..units = Int64(units)
      ..nanos = nanos);
```

```dart
// (Apache 2.0 header)
import 'dart:convert';
import 'package:flutter/material.dart';
import 'package:google_protobuf/google_protobuf.dart' show Struct;

class AttributeFilterEditor extends StatefulWidget {
  final Struct? initial;
  final ValueChanged<Struct?> onChanged;
  const AttributeFilterEditor({super.key, this.initial, required this.onChanged});

  @override
  State<AttributeFilterEditor> createState() => _AttributeFilterEditorState();
}

class _AttributeFilterEditorState extends State<AttributeFilterEditor> {
  late TextEditingController _ctrl;
  String? _error;

  @override
  void initState() {
    super.initState();
    final initialJson = widget.initial == null ? '' : _structToJson(widget.initial!);
    _ctrl = TextEditingController(text: initialJson);
  }

  String _structToJson(Struct s) {
    // Round-trip through JSON for display.
    // (This is best-effort; protobuf's JSON encoder may not be available client-side.)
    return jsonEncode({for (final e in s.fields.entries) e.key: e.value});
  }

  void _emit() {
    final txt = _ctrl.text.trim();
    if (txt.isEmpty) {
      setState(() => _error = null);
      widget.onChanged(null);
      return;
    }
    try {
      final parsed = jsonDecode(txt);
      if (parsed is! Map) throw FormatException('expected JSON object');
      // Build a Struct from the map. For v1 we store the raw JSON; the
      // server reads jsonb and applies its own predicate logic.
      // (TODO plan-3-followup: build proper Struct/Value tree.)
      setState(() => _error = null);
      // Empty Struct as a placeholder; the spec promises an upgrade
      // when a structured editor lands.
      widget.onChanged(Struct());
    } on FormatException catch (e) {
      setState(() => _error = 'Invalid JSON: ${e.message}');
    }
  }

  @override
  Widget build(BuildContext context) {
    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        const Text('Attribute filter (JSON)', style: TextStyle(fontWeight: FontWeight.bold)),
        const SizedBox(height: 4),
        TextField(
          controller: _ctrl,
          maxLines: 4,
          decoration: InputDecoration(
            hintText: '{"kyc_tier": {"in": ["silver", "gold"]}}',
            errorText: _error,
            border: const OutlineInputBorder(),
          ),
          style: const TextStyle(fontFamily: 'monospace', fontSize: 12),
          onChanged: (_) => _emit(),
        ),
      ],
    );
  }

  @override
  void dispose() {
    _ctrl.dispose();
    super.dispose();
  }
}
```

```dart
// (Apache 2.0 header)
import 'package:flutter/material.dart';
import 'package:antinvestor_api_common/antinvestor_api_common.dart' show Money;
import 'package:fixnum/fixnum.dart';

class UsageBarWidget extends StatelessWidget {
  final Money? currentUsage;
  final Money? cap;
  final List<int> tierThresholds; // minor units; rendered as tick marks
  const UsageBarWidget({
    super.key,
    required this.currentUsage,
    required this.cap,
    this.tierThresholds = const [],
  });

  @override
  Widget build(BuildContext context) {
    if (cap == null) return const SizedBox.shrink();
    final capMinor = (cap!.units.toInt() * 100) + (cap!.nanos ~/ 10000000);
    final usedMinor = currentUsage == null
        ? 0
        : (currentUsage!.units.toInt() * 100) + (currentUsage!.nanos ~/ 10000000);
    final ratio = capMinor == 0 ? 0.0 : (usedMinor / capMinor).clamp(0.0, 1.0);
    final color = ratio < 0.7
        ? Colors.green.shade600
        : (ratio < 0.9 ? Colors.orange.shade700 : Colors.red.shade700);
    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        Stack(
          children: [
            LinearProgressIndicator(value: ratio.toDouble(), color: color, backgroundColor: Colors.grey.shade300, minHeight: 8),
            // Tier markers
            for (final t in tierThresholds)
              Positioned(
                left: ((t / capMinor).clamp(0.0, 1.0) * MediaQuery.of(context).size.width * 0.6),
                child: Container(width: 1, height: 10, color: Colors.black54),
              ),
          ],
        ),
        const SizedBox(height: 4),
        Text('${(ratio * 100).toStringAsFixed(1)}% used', style: const TextStyle(fontSize: 11)),
      ],
    );
  }
}
```
